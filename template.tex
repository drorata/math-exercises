\documentclass{minimal}

\usepackage{tikz}
\usetikzlibrary{turtle}

\def\xs{\VAR{num_list}}

\begin{document}

\begin{tikzpicture}[thick, turtle/distance=4em]
  \draw [thick][turtle={home,right}] coordinate (t0)
  \foreach \x [count=\i] in \xs

    [turtle={forward,right}] coordinate (t\i)
   };

\end{tikzpicture}
